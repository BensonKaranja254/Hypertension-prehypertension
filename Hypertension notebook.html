from scipy.stats import shapiro, normaltest, anderson

# Assuming 'data' is your DataFrame containing the dataset and 'Heart' is the column for heart rate
heart_rate_data = data['Heart']

# Shapiro-Wilk Test
shapiro_test_statistic, shapiro_p_value = shapiro(heart_rate_data)
print("Shapiro-Wilk Test:")
print("Test Statistic:", shapiro_test_statistic)
print("p-value:", shapiro_p_value)

# D'Agostino and Pearson's Test
dp_test_statistic, dp_p_value = normaltest(heart_rate_data)
print("\nD'Agostino and Pearson's Test:")
print("Test Statistic:", dp_test_statistic)
print("p-value:", dp_p_value)

# Anderson-Darling Test
anderson_test_statistic, anderson_critical_values, anderson_significance_levels = anderson(heart_rate_data)
print("\nAnderson-Darling Test:")
print("Test Statistic:", anderson_test_statistic)
print("Critical Values:", anderson_critical_values)
print("Significance Levels:", anderson_significance_levels)
